#### String Interpolation

```typescript
import {Component} from '@angular/core';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrl: './app.component.css'
})
export class AppComponent {
    displayedVariable = signal(42);
    valueJustForTs = 'ok';
}
```
<!-- .element: class="fragment" -->

```angular181html
<p> {{ displayedVariable() }} </p>    ==> <p>42</p>
```
<!-- .element: class="fragment" -->

```typescript
this.displayedVariable.set(3)         ==> <p>3</p>
```
<!-- .element: class="fragment" -->

#### Event Binding
```html
<button (click)='foo()'></button>
```
<!-- .element: class="fragment stretch" -->

```typescript
@Component({ ... })
export class AppComponent {
  n = signal(42);
  
  foo() {
    if (this.n() < 100)
      this.n.set(this.n() + 1);
    else
      this.n.update(value => value - 1)
    console.log(this.n())
  }
}
```
<!-- .element: class="fragment stretch" -->

#### Styling
```angular181html
<p [class.hidden]='error()'>Error!</p>
```
<!-- .element: class="fragment" -->

```text
if error():
  give the <hidden> class
```
<!-- .element: class="fragment" -->

```html
<p [style.display]='error() ? "none" : "block"'>Error!</p>
```
<!-- .element: class="fragment" -->

mehrere
<!-- .element: class="fragment" -->
```angular181html
<p [ngClass]='{"hidden": error(), "huge": aria()}'>Error!</p>
```
<!-- .element: class="fragment" -->

```text
if error():
  give the <hidden> class
if aria():
  give the <huge> class
```
<!-- .element: class="fragment" -->
