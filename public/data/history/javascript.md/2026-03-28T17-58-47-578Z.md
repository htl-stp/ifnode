#### `use strict`
```Javascript

"use strict";
```

<ul>
    <li class="fragment fade-in-then-semi-out">sollte die erste Zeile fast jeder .js-Datei sein</li>

#### Typumwandlung
```Javascript
Number(null)        -> 0
Number(undefined)   -> NaN
Number(true)        -> 1
Number(false)       -> 0
Number("")          -> 0
Number("text")      -> NaN
Number("3.1415")    -> 3.1415
"2" / "3"           -> 0.6666666666666666
```

#### Callbacks
Parameter einer Funktion können Funktionen sein
<!-- .element: class="fragment fade-in-then-semi-out" -->

```Javascript
function ifPresent(object, consumer) {
    if (object)
        consumer(object);
}
```
<!-- .element: class="fragment" -->

```Javascript
function log(object) {
    console.log(object);
}
```
<!-- .element: class="fragment" -->

```Javascript
ifPresent(42, log);
```
<!-- .element: class="fragment fade-in-then-semi-out" -->

```Javascript
ifPresent({name: 'callback'},
        function (object) {
        console.log(object);
    });
```
<!-- .element: class="fragment fade-in-then-semi-out" -->

```Javascript
ifPresent({name: 'callback'},
        object => console.log('Present: ', object));
```

#### `for in`
iteriert über member eines objects
<!-- .element: class="fragment fade-in-then-semi-out" -->

```Javascript
let webDev = {
    content: 'html',
    style: 'css',
    action: 'js',
    since: 1995,
    oop: true,
    log() {
        console.log(this);
    }
};
```
<!-- .element: class="fragment" -->

```Javascript
for (let property in webDev)
    console.log(`${property}\t${webDev[property]}`)
```

